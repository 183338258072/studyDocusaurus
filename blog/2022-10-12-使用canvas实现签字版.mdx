function signatureBoard() {
  const SignatureFun = () => {
         const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");
     canvas.addEventListener('mousemove',(e) => {
        const {clientX,clientY} = e;
        ctx.lineTo(clientX, clientY);
        ctx.strokeStyle = 'red'
        ctx.stroke();
        ctx.fill()
     })

    canvas.addEventListener('click',(e) => {
      const {clientX,clientY} = e;
        ctx.beginPath();
        ctx.moveTo(clientX, clientY);
     })
  }
  return (
    <>
        <canvas id='canvas' width='300' height='300' style={{border: '1px solid red'}}> </canvas>
        <button style={{  boxSizing: 'border-box',padding:' 15px 10px',borderRadius: 5,border: 'none'}} onClick={() => SignatureFun()}> 点击开始</button>
    </>

  )
}
