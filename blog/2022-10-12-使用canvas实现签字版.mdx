function signatureBoard() {
  const SignatureFun = () => {
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");
     canvas.addEventListener('mousemove',(e) => {
        const {clientX,clientY} = e;
        ctx.lineTo(clientX, clientY);
        ctx.stroke();
     })

    canvas.addEventListener('mousedown',(e) => {
      const {clientX,clientY} = e;
        ctx.beginPath();
        ctx.moveTo(clientX, clientY);
     })
     
    canvas.addEventListener('mouseup',(e) => {
      const {clientX,clientY} = e;
      ctx.moveTo(clientX, clientY);
    })
  }
  return (
    <div style={{display: 'flex',gap: 20,flexDirection: 'column'}}>
        <canvas id='canvas' width='500' height='500' style={{border: '1px solid red'}}> </canvas>
        <button style={{  boxSizing: 'border-box',padding:' 15px 10px',borderRadius: 5,border: 'none',cursor: 'pointer'}} onClick={() => SignatureFun()}> 点击开始</button>
    </div>

  )
}
